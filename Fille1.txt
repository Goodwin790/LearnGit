git status (позволяет просмотреть статус локального репозитория)
git add название файла (добавляет файл в локальный репозиторий для коммита)
git add . (добавляет все изменённые файлы из локального репозитория)
Git commit -m "сообщение коммита"(подготавливает файл к переносу в гит)
-a (команда объединяющая все файлы локального репозитория)
cat название файла (Команда позволяющая увидеть содержимое файла на локальном репозитории)
git fetch (показывает изменения на удалённом репозитории, которые пока не вступили в силу на локальном репозитории (выполняется перед git pull))
git pull (объединяет данные локального и удалённого репозитория)
git push (финальная команда для завершения коммита)
git log (команда похволяющая увидеть все коммиты и данные о них в локальном и удалённом репозиториях)(для выхода из команды нажать "q")
git log --author имя автора (показывает все изменения сделанные данным автором)
git show (показывает данные о последнем сделанном коммите)
git show хэшкод коммита (показывает все данные об изменениях в указанном коммите)
git blame название файла (выводит информацию о том, какие изменения вносил тот или иной автор)
git blame название файла | grep интересующая строка (выводит информацию об авторе изменявшем данную строку и времени когда она была изменена)
git blame название файла | grep имя автора (выводит все изменения сделанные данным автором в данном файле)
git diff (отображает изменения в коммитах)
git echo "добавляемая информация" >> имя файла (добавляет изменения в указанный файл)
